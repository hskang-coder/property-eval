/** * ========================================================= * 물건 경쟁력 평가 — 구글 시트 저장 웹앱 * ========================================================= * * [설치 방법 — 5분] * * 1. 저장할 구글 스프레드시트를 새로 만들거나 기존 것 열기 * 2. 상단 메뉴 → 확장 프로그램 → Apps Script * 3. 이 코드 전체 붙여넣고 저장 (Ctrl+S) * 4. 상단 "배포" 버튼 → "새 배포" * 5. 유형: 웹 앱 선택 * - 설명: 물건경쟁력 평가 수신 * - 다음 사용자로 실행: 나 (본인 계정) * - 액세스 권한: 모든 사용자 (익명 포함) * 6. "배포" 클릭 → 권한 허용 * 7. 표시되는 "웹 앱 URL" 복사 * 8. 평가폼 HTML 파일 맨 위 SCRIPT_URL 에 붙여넣기 * * ※ 코드 수정 후에는 "배포 관리 → 새 버전"으로 재배포 필요 * ========================================================= */ const SHEET_NAME = '물건경쟁력평가'; // 저장할 시트 이름 (없으면 자동 생성) // 열 순서 정의 const COLUMNS = [ '제출시각', '계약번호', '임차인명', '호실', '기록담당자', '방문일', // 주변환경 '역거리', '언덕여부', '주변안전감', '경쟁밀집도', '고시원느낌', // 건물외부 '외관상태', '낙서오염', '우편함공용부', '입구안전성', // 건물내부 '보안시설', '복도계단', '관리상태', '엘리베이터', // 방내부 '방크기', '채광', '도배장판', '화장실', '주방가구', '가전', // 보정값 '중도퇴실반복', '주변시세', '층수', // 환산점수 '주변환경점수', '건물외부점수', '건물내부점수', '방내부점수', '퇴실반복점수', '시세점수', '층수점수', // 결과 '총점', '등급', '경보' ]; // ── CORS 프리플라이트 대응 ────────────────────────────────── function doOptions(e) { return ContentService .createTextOutput('') .setMimeType(ContentService.MimeType.TEXT); } // ── POST 수신 ────────────────────────────────────────────── function doPost(e) { try { const raw = e.postData.contents; const data = JSON.parse(raw); const sheet = getOrCreateSheet(); // 헤더가 없으면 첫 행에 작성 if (sheet.getLastRow() === 0) { sheet.appendRow(COLUMNS); styleHeader(sheet); } // 데이터 행 구성 (COLUMNS 순서대로, 없는 키는 빈 문자열) const row = COLUMNS.map(col => data[col] !== undefined ? data[col] : ''); sheet.appendRow(row); // 등급에 따라 행 색상 적용 applyRowColor(sheet, sheet.getLastRow(), data['등급']); return ContentService .createTextOutput(JSON.stringify({ status: 'ok' })) .setMimeType(ContentService.MimeType.JSON); } catch (err) { return ContentService .createTextOutput(JSON.stringify({ status: 'error', message: err.message })) .setMimeType(ContentService.MimeType.JSON); } } // ── GET 요청 — 연결 테스트용 ─────────────────────────────── function doGet() { return ContentService .createTextOutput('물건경쟁력 평가 수신 서버 정상 작동 중') .setMimeType(ContentService.MimeType.TEXT); } // ── 시트 가져오기 / 생성 ─────────────────────────────────── function getOrCreateSheet() { const ss = SpreadsheetApp.getActiveSpreadsheet(); let sheet = ss.getSheetByName(SHEET_NAME); if (!sheet) { sheet = ss.insertSheet(SHEET_NAME); } return sheet; } // ── 헤더 스타일 ──────────────────────────────────────────── function styleHeader(sheet) { const headerRange = sheet.getRange(1, 1, 1, COLUMNS.length); headerRange .setBackground('#263238') .setFontColor('#FFFFFF') .setFontWeight('bold') .setFontSize(11); sheet.setFrozenRows(1); sheet.autoResizeColumns(1, COLUMNS.length); } // ── 등급별 행 색상 ───────────────────────────────────────── function applyRowColor(sheet, rowNum, grade) { const colors = { '최상급': '#E3F2FD', '상급' : '#E8F5E9', '일반급': '#FFFDE7', '하급' : '#FFF3E0', '최하급': '#FFEBEE', }; const color = colors[grade]; if (color) { sheet.getRange(rowNum, 1, 1, COLUMNS.length).setBackground(color); } // 총점·등급 열 굵게 const totalColIdx = COLUMNS.indexOf('총점') + 1; const gradeColIdx = COLUMNS.indexOf('등급') + 1; if (totalColIdx > 0) sheet.getRange(rowNum, totalColIdx).setFontWeight('bold'); if (gradeColIdx > 0) sheet.getRange(rowNum, gradeColIdx).setFontWeight('bold'); }
